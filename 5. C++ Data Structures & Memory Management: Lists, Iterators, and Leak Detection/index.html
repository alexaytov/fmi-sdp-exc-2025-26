template <typename T>
class DoublyLinkedList {
private:
    Node<T>* head_;
    Node<T>* tail_;
    unsigned int size_;

public:
    DoublyLinkedList() : head_(nullptr), tail_(nullptr), size_(0) {}
    
    ~DoublyLinkedList() {

    }
};
