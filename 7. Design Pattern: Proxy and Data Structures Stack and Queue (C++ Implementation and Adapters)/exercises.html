        // –í–∑–µ–º–∏ —Å–ª–µ–¥–≤–∞—â–∞—Ç–∞ –æ—Ç forward stack
        currentPage = forwardStack.top();
        forwardStack.pop();
        
        std::cout << "Went forward to: " << currentPage->getURL() << std::endl;
    }
    
    // –ü–æ–∫–∞–∂–∏ —Ç–µ–∫—É—â–∞—Ç–∞ —Å—Ç—Ä–∞–Ω–∏—Ü–∞
    void displayCurrent() {
        if (currentPage == nullptr) {
            std::cout << "No page loaded" << std::endl;
            return;
        }
        
        std::string url = currentPage->getURL();
        
        // –û–ø–∏—Ç–∞–π –æ—Ç cache
        if (cache.has(url)) {
            std::cout << "[CACHED] " << cache.get(url) << std::endl;
        } else {
            std::string content = currentPage->getContent();
            std::cout << content << std::endl;
        }
    }
    
    ~BrowserNavigator() {
        delete currentPage;
        while (!backStack.empty()) {
            delete backStack.top();
            backStack.pop();
        }
        while (!forwardStack.empty()) {
            delete forwardStack.top();
            forwardStack.pop();
        }
    }
};
