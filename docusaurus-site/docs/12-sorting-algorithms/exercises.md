    // –†–∞–∑–º–µ—Ä–∏ –Ω–∞ –¥–≤–∞—Ç–∞ –ø–æ–¥–º–∞—Å–∏–≤–∞

        L[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        R[j] = arr[mid + 1 + j];

    // Merge –Ω–∞ –≤—Ä–µ–º–µ–Ω–Ω–∏—Ç–µ –º–∞—Å–∏–≤–∏ –æ–±—Ä–∞—Ç–Ω–æ –≤ arr[left..right]

    int j = 0;    // –ù–∞—á–∞–ª–µ–Ω –∏–Ω–¥–µ–∫—Å –Ω–∞ –≤—Ç–æ—Ä–∏—è –ø–æ–¥–º–∞—Å–∏–≤

            arr[k] = L[i];
            i++;
        } else {
            arr[k] = R[j];
            j++;
        }
        k++;
    }

    // –ö–æ–ø–∏—Ä–∞–Ω–µ –Ω–∞ –æ—Å—Ç–∞–Ω–∞–ª–∏—Ç–µ –µ–ª–µ–º–µ–Ω—Ç–∏ –æ—Ç L[], –∞–∫–æ –∏–º–∞

        arr[k] = L[i];
        i++;
        k++;

    while (j < n2) {
        arr[k] = R[j];
        j++;
        k++;
    }
}

// Insertion Sort –∑–∞ –º–∞–ª–∫–∏ –ø–æ–¥–º–∞—Å–∏–≤–∏
void insertionSort(std::vector<int>& arr, int left, int right) {
    for (int i = left + 1; i <= right; i++) {
        int key = arr[i];
        int j = i - 1;

        while (j >= left && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }

// Merge —Ñ—É–Ω–∫—Ü–∏—è
void merge(std::vector<int>& arr, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    std::vector<int> L(n1), R(n2);

    for (int i = 0; i < n1; i++)
        L[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        R[j] = arr[mid + 1 + j];

    int i = 0, j = 0, k = left;

    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k++] = L[i++];
        } else {
            arr[k++] = R[j++];
        }
    }

    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];

void hybridMergeSort(std::vector<int>& arr, int left, int right) {

        int mid = left + (right - left) / 2;

        hybridMergeSort(arr, left, mid);
        hybridMergeSort(arr, mid + 1, right);

}

void bucketSort(std::vector<float>& arr) {
    int n = arr.size();
    if (n <= 0) return;

    // –°—ä–∑–¥–∞–≤–∞–Ω–µ –Ω–∞ 10 –ø—Ä–∞–∑–Ω–∏ buckets
    const int NUM_BUCKETS = 10;
    std::vector<std::vector<float>> buckets(NUM_BUCKETS);

    // –†–∞–∑–ø—Ä–µ–¥–µ–ª—è–Ω–µ –Ω–∞ –µ–ª–µ–º–µ–Ω—Ç–∏—Ç–µ –≤ buckets
    for (int i = 0; i < n; i++) {
        int bucketIndex = static_cast<int>(arr[i] * NUM_BUCKETS);

        // Handle edge case: arr[i] == 1.0
        if (bucketIndex == NUM_BUCKETS) {
            bucketIndex = NUM_BUCKETS - 1;
        }

        buckets[bucketIndex].push_back(arr[i]);
    }

    // –°–æ—Ä—Ç–∏—Ä–∞–Ω–µ –Ω–∞ –≤—Å–µ–∫–∏ bucket —Å insertion sort
    for (int i = 0; i < NUM_BUCKETS; i++) {
        // Insertion sort –Ω–∞ bucket
        for (size_t j = 1; j < buckets[i].size(); j++) {
            float key = buckets[i][j];
            int k = j - 1;

            while (k >= 0 && buckets[i][k] > key) {
                buckets[i][k + 1] = buckets[i][k];
                k--;
            }
            buckets[i][k + 1] = key;
        }
    }

    // –ö–æ–Ω–∫–∞—Ç–µ–Ω–∏—Ä–∞–Ω–µ –Ω–∞ buckets –æ–±—Ä–∞—Ç–Ω–æ –≤ arr
    int index = 0;
    for (int i = 0; i < NUM_BUCKETS; i++) {
        for (size_t j = 0; j < buckets[i].size(); j++) {
            arr[index++] = buckets[i][j];
        }
    }
}

// Bubble Sort
void bubbleSort(std::vector<int>& arr) {
    int n = arr.size();
    for (int i = 0; i < n - 1; ++i) {
        bool swapped = false;
        for (int j = 0; j < n - i - 1; ++j) {
            if (arr[j] > arr[j + 1]) {
                std::swap(arr[j], arr[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

// Insertion Sort
void insertionSort(std::vector<int>& arr) {
    int n = arr.size();
    for (int i = 1; i < n; ++i) {
        int key = arr[i];
        int j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}
