### Iterator Pattern - –û—Å–Ω–æ–≤–Ω–∏ –û–ø–µ—Ä–∞—Ü–∏–∏

```cpp
class LinkedListIterator {
private:
    Node<T>* current;

public:
    LinkedListIterator(Node<T>* ptr) : current(ptr) {}

    // –î–µ—Ä–µ—Ñ–µ—Ä–µ–Ω—Ü–∏—Ä–∞–Ω–µ - –¥–æ—Å—Ç—ä–ø –¥–æ —Å—Ç–æ–π–Ω–æ—Å—Ç—Ç–∞

    LinkedListIterator& operator++() {
        current = current->next;
        return *this;
    }

    // Post-increment
    LinkedListIterator operator++(int) {
        LinkedListIterator temp(*this);
        current = current->next;
        return temp;
    }

    // Pre-decrement (–∑–∞ DLL)
    LinkedListIterator& operator--() {
        current = current->prev;
        return *this;
    }

    // –°—Ä–∞–≤–Ω–µ–Ω–∏–µ
    bool operator==(const LinkedListIterator& other) const {
        return current == other.current;
    }

    bool operator!=(const LinkedListIterator& other) const {
        return current != other.current;
    }
};
```

### 6.1. –ü—Ä–∏–º–µ—Ä–Ω–∞ –ò–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è –Ω–∞ –î–≤—É—Å–≤—ä—Ä–∑–∞–Ω –°–ø–∏—Å—ä–∫

```cpp
template <typename T>
class Node {
public:
    T data;
    Node<T>* next;
    Node<T>* prev;

    Node() : next(nullptr), prev(nullptr) {}
    Node(const T& item, Node<T>* n = nullptr, Node<T>* p = nullptr)
        : data(item), next(n), prev(p) {}
};

template <typename T>
class DoublyLinkedList {
private:
    Node<T>* head_;
    Node<T>* tail_;
    unsigned int size_;

public:
    DoublyLinkedList() : head_(nullptr), tail_(nullptr), size_(0) {}

    ~DoublyLinkedList() {
        destroy_list();
    }

    void destroy_list() {
        Node<T>* current = head_;
        while (current != nullptr) {
            Node<T>* temp = current->next;
            delete current;
            current = temp;
        }
        head_ = nullptr;
        tail_ = nullptr;
        size_ = 0;
    }

    void push_back(const T& val) {
        Node<T>* newnode = new Node<T>(val);

        if (!tail_) { // –ü—Ä–∞–∑–µ–Ω —Å–ø–∏—Å—ä–∫
            head_ = newnode;
            tail_ = newnode;
        } else {
            tail_->next = newnode;
            newnode->prev = tail_;
            tail_ = newnode;
        }
        size_++;
    }

    unsigned int size() const { return size_; }

    // Iterator typedef
    typedef DoublyLinkedListIterator<T> iterator;

    iterator begin() { return iterator(head_); }
    iterator end() { return iterator(nullptr); }
};
```
