        int v = s.top();
        s.pop();

        if (!visited[v]) {
            visited[v] = true;
            cout << v << " ";

            // –î–æ–±–∞–≤—è–º–µ —Å—ä—Å–µ–¥–∏—Ç–µ –≤ —Å—Ç–µ–∫–∞
            // –û–±—ä—Ä–Ω–∞—Ç —Ä–µ–¥ –∑–∞ –¥–∞ —Å—ä–æ—Ç–≤–µ—Ç—Å—Ç–≤–∞ –Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–∞—Ç–∞ –≤–µ—Ä—Å–∏—è
            for (auto it = adj[v].rbegin(); it != adj[v].rend(); ++it) {

### –ü—Ä–∏–º–µ—Ä 1: –°–æ—Ü–∏–∞–ª–Ω–∞ –ú—Ä–µ–∂–∞ - –ü—Ä–∏—è—Ç–µ–ª–∏ –Ω–∞ k –°—Ç–µ–ø–µ–Ω–∏

// –ù–∞–º–∏—Ä–∞–Ω–µ –Ω–∞ –≤—Å–∏—á–∫–∏ –ø—Ä–∏—è—Ç–µ–ª–∏ –Ω–∞ —Ä–∞–∑—Å—Ç–æ—è–Ω–∏–µ –¥–æ k –æ—Ç start
vector<int> friendsWithinK(const vector<vector<int>>& adj, int start, int k) {
    vector<int> result;
    vector<int> distance(adj.size(), -1);
    queue<int> q;

    distance[start] = 0;
    q.push(start);

    while (!q.empty()) {
        int v = q.front();
        q.pop();

        if (distance[v] <= k) {
            if (v != start) result.push_back(v);

            for (int neighbor : adj[v]) {
                if (distance[neighbor] == -1) {
                    distance[neighbor] = distance[v] + 1;
                    q.push(neighbor);
                }
            }
        }
    }

    return result;
}
```

### –ü—Ä–∏–º–µ—Ä 2: –û—Ç–∫—Ä–∏–≤–∞–Ω–µ –Ω–∞ –¶–∏–∫—ä–ª –≤ –ù–∞—Å–æ—á–µ–Ω –ì—Ä–∞—Ñ

bool hasCycleDFS(const vector<vector<int>>& adj, int v,
                 vector<int>& color) {
    // color: 0 = white (unvisited), 1 = gray (visiting), 2 = black (done)
    color[v] = 1; // gray

    for (int neighbor : adj[v]) {
        if (color[neighbor] == 1) {
            // Back edge found - —Ü–∏–∫—ä–ª!
            return true;
        }
        if (color[neighbor] == 0 && hasCycleDFS(adj, neighbor, color)) {
            return true;
        }
    }

    color[v] = 2; // black
    return false;
}

bool hasCycle(const vector<vector<int>>& adj) {
    int n = adj.size();
    vector<int> color(n, 0);

    for (int i = 0; i < n; i++) {
        if (color[i] == 0) {
            if (hasCycleDFS(adj, i, color)) {
                return true;
            }
        }
    }
    return false;
}
```

void DFS(const vector<vector<int>>& adj, int v, vector<bool>& visited) {
    visited[v] = true;
    for (int neighbor : adj[v]) {
        if (!visited[neighbor]) {
            DFS(adj, neighbor, visited);
        }
    }
}

int countComponents(const vector<vector<int>>& adj) {
    int n = adj.size();
    vector<bool> visited(n, false);
    int count = 0;

    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            DFS(adj, i, visited);
            count++;
        }
    }

    return count;
}
