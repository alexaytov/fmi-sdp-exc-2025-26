    // –†–∞–∑–º—è–Ω–∞ –Ω–∞ –ø–æ–ª–æ–≤–∏–Ω–∏—Ç–µ

    return num;
}

### –ó–∞–¥–∞—á–∞ 24: –£–º–Ω–æ–∂–µ–Ω–∏–µ —Å –ø–æ–±–∏—Ç–æ–≤–∏ –æ–ø–µ—Ä–∞—Ü–∏–∏

### –ó–∞–¥–∞—á–∞ 25: BitSet –∫–ª–∞—Å –∑–∞ —Ñ–ª–∞–≥–æ–≤–µ

–°—ä–∑–¥–∞–π—Ç–µ –∫–ª–∞—Å `BitSet`, –∫–æ–π—Ç–æ —Å—ä—Ö—Ä–∞–Ω—è–≤–∞ –¥–æ 32 –±—É–ª–µ–≤–∏ —Ñ–ª–∞–≥–∞ –≤ –µ–¥–∏–Ω `unsigned int` –∏ –ø—Ä–µ–¥–æ—Å—Ç–∞–≤—è –º–µ—Ç–æ–¥–∏:

- `void clear(int pos)` - –Ω—É–ª–∏—Ä–∞ –±–∏—Ç
- `bool test(int pos)` - –ø—Ä–æ–≤–µ—Ä—è–≤–∞ –±–∏—Ç

**–†–µ—à–µ–Ω–∏–µ:**

```cpp

private:
    unsigned int bits;

public:
    // –ö–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä
    BitSet() : bits(0) {}

    // –£—Å—Ç–∞–Ω–æ–≤—è–≤–∞–Ω–µ –Ω–∞ –±–∏—Ç
    void set(int pos) {
        if (pos < 0 || pos >= 32) {
            throw std::out_of_range("Position out of range");
        }
        bits |= (1u << pos);
    }

    // –ù—É–ª–∏—Ä–∞–Ω–µ –Ω–∞ –±–∏—Ç
    void clear(int pos) {
        if (pos < 0 || pos >= 32) {
            throw std::out_of_range("Position out of range");
        }
        bits &= ~(1u << pos);
    }

    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –±–∏—Ç
    bool test(int pos) const {
        if (pos < 0 || pos >= 32) {
            throw std::out_of_range("Position out of range");
        }
        return (bits & (1u << pos)) != 0;
    }

    // –ò–Ω–≤–µ—Ä—Ç–∏—Ä–∞–Ω–µ –Ω–∞ –±–∏—Ç
    void toggle(int pos) {
        if (pos < 0 || pos >= 32) {
            throw std::out_of_range("Position out of range");
        }
        bits ^= (1u << pos);
    }

    // –ë—Ä–æ–µ–Ω–µ –Ω–∞ —É—Å—Ç–∞–Ω–æ–≤–µ–Ω–∏ –±–∏—Ç–æ–≤–µ
    int count() const {
        int cnt = 0;
        unsigned int temp = bits;
        while (temp) {
            temp &= (temp - 1);  // –ü—Ä–µ–º–∞—Ö–≤–∞ –Ω–∞–π-–¥–µ—Å–Ω–∏—è –µ–¥–∏–Ω–∏—á–µ–Ω –±–∏—Ç
            cnt++;
        }
        return cnt;
    }

    // –î–æ–ø—ä–ª–Ω–∏—Ç–µ–ª–Ω–∏ –ø–æ–ª–µ–∑–Ω–∏ –º–µ—Ç–æ–¥–∏

    // –£—Å—Ç–∞–Ω–æ–≤—è–≤–∞–Ω–µ –Ω–∞ –≤—Å–∏—á–∫–∏ –±–∏—Ç–æ–≤–µ
    void setAll() {
        bits = ~0u;
    }

    // –ù—É–ª–∏—Ä–∞–Ω–µ –Ω–∞ –≤—Å–∏—á–∫–∏ –±–∏—Ç–æ–≤–µ
    void clearAll() {
        bits = 0;
    }

    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ –ø–æ–Ω–µ –µ–¥–∏–Ω –±–∏—Ç –µ —É—Å—Ç–∞–Ω–æ–≤–µ–Ω
    bool any() const {
        return bits != 0;
    }

    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ –≤—Å–∏—á–∫–∏ –±–∏—Ç–æ–≤–µ —Å–∞ –Ω—É–ª–∏
    bool none() const {
        return bits == 0;
    }

    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ –≤—Å–∏—á–∫–∏ –±–∏—Ç–æ–≤–µ —Å–∞ –µ–¥–∏–Ω–∏—Ü–∏
    bool all() const {
        return bits == ~0u;
    }

    // –†–∞–∑–º–µ—Ä (–≤–∏–Ω–∞–≥–∏ 32 –∑–∞ —Ç–æ–∑–∏ –∫–ª–∞—Å)
    int size() const {
        return 32;
    }

    // –î–æ—Å—Ç—ä–ø –¥–æ –≤—ä—Ç—Ä–µ—à–Ω–æ—Ç–æ –ø—Ä–µ–¥—Å—Ç–∞–≤—è–Ω–µ
    unsigned int value() const {
        return bits;
    }

    // –ü—Ä–∏–Ω—Ç–∏—Ä–∞–Ω–µ –≤ –¥–≤–æ–∏—á–Ω–∞ —Ñ–æ—Ä–º–∞
    void print() const {
        for (int i = 31; i >= 0; i--) {
            std::cout << ((bits & (1u << i)) ? '1' : '0');
            if (i % 8 == 0) std::cout << ' ';
        }
        std::cout << '\n';
    }

    // Overload –Ω–∞ –æ–ø–µ—Ä–∞—Ç–æ—Ä []
    class BitReference {
    private:
        BitSet& bs;
        int pos;
    public:
        BitReference(BitSet& b, int p) : bs(b), pos(p) {}

        // –ü—Ä–∏—Å–≤–æ—è–≤–∞–Ω–µ
        BitReference& operator=(bool value) {
            if (value) {
                bs.set(pos);
            } else {
                bs.clear(pos);
            }
            return *this;
        }

        // –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–∞–Ω–µ –∫—ä–º bool
        operator bool() const {
            return bs.test(pos);
        }
    };

    BitReference operator[](int pos) {
        return BitReference(*this, pos);
    }

    bool operator[](int pos) const {
        return test(pos);
    }
};
```
