```cpp

// –°—Ç—Ä—É–∫—Ç—É—Ä–∞ –Ω–∞ –≤—ä–∑–µ–ª –≤ –¥–≤—É—Å–≤—ä—Ä–∑–∞–Ω —Å–ø–∏—Å—ä–∫

    // –ö–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä

    // –î–µ—Å—Ç—Ä—É–∫—Ç–æ—Ä –∑–∞ –æ—Å–≤–æ–±–æ–∂–¥–∞–≤–∞–Ω–µ –Ω–∞ –ø–∞–º–µ—Ç—Ç–∞

        Node* current = head;
        while (current != nullptr) {
            Node* next_node = current->next;
            delete current;
            current = next_node;
        }
        head = nullptr;
        tail = nullptr;
    }

    // –î–æ–±–∞–≤—è–Ω–µ –Ω–∞ –µ–ª–µ–º–µ–Ω—Ç –≤ –∫—Ä–∞—è –Ω–∞ —Å–ø–∏—Å—ä–∫–∞

        if (tail) { // –ê–∫–æ —Å–ø–∏—Å—ä–∫—ä—Ç –Ω–µ –µ –ø—Ä–∞–∑–µ–Ω
            tail->next = newNode;
            newNode->prev = tail;
            tail = newNode;
        } else { // –ê–∫–æ —Å–ø–∏—Å—ä–∫—ä—Ç –µ –ø—Ä–∞–∑–µ–Ω
            head = newNode;
            tail = newNode;
        }
    }

    // –ü—Ä–µ–º–∞—Ö–≤–∞–Ω–µ –Ω–∞ –∫–æ–Ω–∫—Ä–µ—Ç–µ–Ω –≤—ä–∑–µ–ª
    void remove(Node* node_to_remove) {
        if (!node_to_remove) return;

        if (node_to_remove->prev) {
            node_to_remove->prev->next = node_to_remove->next;
        } else { // –ê–∫–æ –ø—Ä–µ–º–∞—Ö–≤–∞–º–µ head
            head = node_to_remove->next;
        }

        if (node_to_remove->next) {
            node_to_remove->next->prev = node_to_remove->prev;
        } else { // –ê–∫–æ –ø—Ä–µ–º–∞—Ö–≤–∞–º–µ tail
            tail = node_to_remove->prev;
        }

        delete node_to_remove; // –û—Å–≤–æ–±–æ–∂–¥–∞–≤–∞–Ω–µ –Ω–∞ –ø–∞–º–µ—Ç—Ç–∞
    }

    // –û–±—Ö–æ–∂–¥–∞–Ω–µ –∏ –∏–∑–≤–µ–∂–¥–∞–Ω–µ –Ω–∞ –µ–ª–µ–º–µ–Ω—Ç–∏—Ç–µ
    void display() const {
        Node* current = head;
        while (current != nullptr) {
            std::cout << current->data << " ";
            current = current->next;
        }
        std::cout << std::endl;
    }

    Node* getHead() const { return head; }
```
